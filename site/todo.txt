Fix same band multiple results

WITH event_band_counts AS (SELECT COUNT(*) as count, contest_event_id, band_id
FROM contest_result
GROUP BY contest_event_id, band_id)
SELECT e.date_of_event, c.slug, bc.band_id, b.name, CONCAT('https://brassbandresults.co.uk/contests/', c.slug, '/', e.date_of_event) as url
FROM contest_event e
INNER JOIN contest c ON c.id = e.contest_id
INNER JOIN event_band_counts bc ON bc.contest_event_id = e.id
INNER JOIN band b ON b.id = bc.band_id
WHERE bc.count > 1
ORDER BY b.name;
